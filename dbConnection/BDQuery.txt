CREATE DATABASE Asignacion3;

USE Asignacion3;

CREATE TABLE Pais (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Nombre VARCHAR(100),
    Gentilicio VARCHAR(100),
    Capital VARCHAR(50),
    Estatus VARCHAR(50)
);

CREATE TABLE Usuario (
	Id INT IDENTITY(1,1) PRIMARY KEY,
	Nombre VARCHAR(100),
	Correo VARCHAR(100),
	Clave VARCHAR(75),
	Estatus VARCHAR(50)
);
